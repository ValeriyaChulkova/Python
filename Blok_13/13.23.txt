prices = [10.5, 7, 15, 12, 9, 8, 11, 13.5, 17, 6, 14, 7.5, 16, 5, 6.5]
editions = [5000, 8000, 10000, 2000, 6000, 4000, 3000, 15000, 9000, 12000, 1000, 7000, 11000, 2500, 30000]

total_price = 0
count = 0
for i in range(len(prices)):
    if editions[i] < 10000:
        total_price += prices[i]
        count += 1
average_price = total_price / count
print('Average price of magazines with edition less than 10000:', average_price)
